#### 题目代码

seed = 0x00000000 # you need to solve this
flag = 'n1book{%x}' % seed

state = seed
mask = 0b10000000000000000000000001010111

def lfsr():
    global state, mask
    output = state & 1
    now = state & mask
    new = 0
    while now:
        new ^= now & 1
        now >>= 1
    state = (new << 31) | (state >> 1)
    return output


for i in range(32):
    lfsr()

print '%x' % state # 155a796b
